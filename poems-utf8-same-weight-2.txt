خلق را از کارشان بايد شناخت
رو که اين خويشي نمي‌آيد بکار
هر چه بود آن شير و اين روباه بود
فيل مي‌بالد به خرطوم دراز
خلقت و تقدير با هم کرده‌اند
گفت در پاسخ رفوگر کاي رفيق
ما که عمري پرده‌داري کرده‌ايم
بردباري مور را افکند و کشت
توشه بستم از حلال و از حرام
گر بچشم خويش بينم مرگ را
روز سائل بود و شب بيمار دار
تو مرا ديوانه خواني اي فلان
کجروان را حق نميگردد دليل
از تنور خودپسندي شد بلند
کوها کارست در اين کارگاه
بر گشائي چشم خواب آلود را
هر چه باشد باز نامش مسند است
گفت بيرون آر دست از آستين
بار هر کس در خور ياراي اوست
خار آن گل ديد و رو در هم کشيد
تيغ ذلت ناخنش کوتاه کرد
خشم هم بر کوزه هم بر آب کرد
من نمي‌بينم ترا در زير پاي
خاک و باد و آب سرگردان ماست
ره نمي‌پرسيم اما ميرويم
که در اندازي مرا هم ناگهان
نيست گل را فرصت بيم و اميد
روز مهماني و بازي شاه بود
از کجا آن لانه را مي‌ساختيم
تو نميداني که از بهر خزان
آگهم کاز بند من بيرون نشست
در چه معبد معبد يزدان پاک
که بجويد گمشده پيوند را
در سراي عمر تعميري کني
دور افتاده ز بزم يارها
يک شبي آن سفله بي ننگ و نام
طرف گلشن را منظم کرده‌اند
چون رهي زين کشتي بي ناخداي
گوهر کان هوي جز سنگ نيست
گر ز يک کنجم براند روزگار
هست بازاري دگر اي خواجه تاش
روز و شب بيهوده سوزن ميزني
زر فزون ميگيرد و کم ميخرد
دام محکم ضعف در حد کمال
ديده تا ياراي ديدن داشت ديد
هر زمان ره ميزند دزد هوي
سر بگرديد از غم و دل شد تباه
گر خريداري در آي اندر دکان
برگرفتي زادمي چون ديو روي
هم عسل هم شوربا را ريختي
پيش من خوردند مردم نان گرم
خيرگي بين خار ناهموار را
هم نظربازان بر آن بگذرند
ميروم من يک دو روز اينجا بمان
آن دل فارغ ز خون آکنده شد
نان نخوردي خاک خوردي اي عجب
از کجا دامان تو آلوده‌ايم
جنس ما را نيست خرد و سالخورد
هر گل سرخي گلستاني شده
قاضي کشمر ز محضر شامگاه
دايره صد جا ولي پرگار نه
قطره را گفتم بدان جانب مريز
بانگ زد خادم بر او کي خود پرست
ليک من عاقلترم از عاقلان
ساعتي جولا زماني بندباز
فيل را آن مور از دنبال رفت
کارفرمايان عالم کرده‌اند
خوابگاه اندر سر ره ساختي
قدرت و ياري از او يارا ز ما
گفت اين خويشان وبال گردنند
آنچه در کيسه است در دامان برد
من چه ميدانم که دستم سوخته است
هر که شاگرد طمع شد دزد شد
جويباري بودي و جيحون شدي
بر نمي آورد بزاز دغل
گفتمش لختي بمان بهر رفو
کار خود ميکن ترا با ما چکار
ما سگ مسکين بازاري نه‌ايم
من هدف بودم قضا را سالها
چرخ و انجم هستي ما ميبرند
قصه‌هاي دل فزون از گفتن است
گر سحر ويران کنند اين سقف و بام
لانه نزديک است از من دور شو
هيچ شد هم پرنيان و هم پلاس
باز کن وقت خريدن چشم را
شب ز اشکش غرق در گوهر کنم
زان سبب شد صيد روباه فلک
اوفتاد آهسته و غلتيد و رفت
گفت فرصت نيست وقت رفتن است
آب خاکت را دهد ناگه بباد
جهد را بسيار کن عمر اندکي است
گر گلم ريزند بر سر دور نيست
هم تو خوردي گاه پخته گاه خام
فرقها بود اين گره را زان گره
گر زيانش ده بود سودش صد است
شاهد بخت است و در پهلوي تست
از براي ما درين بحر عميق
نيکيم را بد شمرد آن سست مهر
خانه درويش از دزد ايمن است
اين چنين رفتن بچاه افتادن است
پيلي از موران نيايد مور شو
خانه تدبير را بر هم زدي
اندرين بزم طرب گوئي ترا
از کريمان از چه رو کم خواستي
چون براي جستجو خم کرد سر
تو فرومايه شدي ضرب‌المثل
صبحدم صاحبدلي در گلشني
آز دزد است و ربودن کار اوست
که تو يکروزي بسوزي در شرار
کس نخواهد گفت کشميري بباف
خانه ما را نميکردي پسند
عقل فرسوده است و در فکر سريم
فرشها کو کيسه‌هاي زر کجاست
کس ز خلقان تو پيراهن نکرد
هرگز اين سود و زيانرا نشمريم
هر که بيني رشته‌اي دارد بدست
عيبها گفتند از هم بيشمار
قفل مخزن را که ديشب ميشکست
هم در افتاديم و هم برخاستيم
گنجها داري و هستي تنگدست
روزها چون جبه‌اش در بر کنم
عاريت ميخواستندش کودکان
کارگر مي خواست زيرا کار بود
من همي خندم برسم روزگار
طاقتي در لنگر و سکان نماند
وقت غير از خوان يغمائي نبود
وان گره بگشوده گندم ريخته
دامن پاکيزه را بر چيد و رفت
چون نمي‌بيند چو تو بيننده‌اي
سنگ را گفتم بزيرش نرم شو
درزي ايام زان ره ميشکافت
چون تو اندر گوشه عزلت نماند
تو چه مي‌بيني به جز وهم و خيال
همچو مغز خالص بي پوستي
واگذار اين لاشه ناچيز را
من بهر جائي که مسکن ميکنم
زر کاني را چه نسبت با سفال
شب لحافش سايه و روز آفتاب
خانه هر کس براي او سزاست
وز شراري خانمان‌ها سوخته
خدمت محضر ز من نايد دگر
خواست تا لاف خداوندي زند
دکه خر مهره جاي ديگر است
آن گل پژمرده چيد و شد روان
گور خود با نوک سوزن کندن است
گلبني کاندر بياباني شکفت
بايدت اندرز ما آموختن
اي عجب امروزها ديروز شد
قفل از صندوق آهن ميگشود
گر ز وي روزي قفائي خورد خورد
گاه بر چيند ز بامي دانه‌اي
آنچه برديم از تو باز آريم باز
رهروان کفش و کلاهت ميبرند
بشکنيم از جهل و خود را نشکنيم
کيست کاز جور قضا آواره نيست
در خرابيهاي ما معمار باش
گر ببيني آن هوا و آن نسيم
همچو شبنم در گلستان وجود
ما نميگوئيم سائل در مزن
غير ازين چيزي بما نفروختند
دوش يک من هيمه را باري نوشت
فيل گفت اين راه مشکل واگذار
دوش طفلان بر سرت گل ريختند
بانگ زد کاي خواجه صبحت خير باد
ديد گلهاي سپيد و سرخ و زرد
ما ز بيداد تو سرگردان شديم
رشته‌ها رشتند در دوک عناد
با چه نيرو بر هوي غالب شويم
ما که دشمن را چنين ميپروريم
رهرو ما اينک اندر منزل است
ما ز يک اقليم زان با هم خوشيم
در رواج کار خود چون من بکوش
گر چه اندر کنج عزلت ساکنم
دخمه بسيار است اين ويرانه را
کاش مي‌دانست روبه ناشتاست
دوستان را از نظر چون ميبريم
اين غريق خرد بهر غرق نيست
راست گفتي آنچه گفتي راست راست
تيشه زان بر هر رگ و بندم زنند
بيني از انديشه خالي عالمي
او نشست آسوده و خفتيم ما
کاندر آنجا مي‌شناسند اين قماش
باخرد صلحي کن و رائي بزن
آن عدس در شوربا ميريختم
آنکه با نمرود اين احسان کند
تو بسي زانديشه برتر بوده‌اي
خوي کرده با جفاي خارها
بس رفو کردم ندانستم که عمر
در ره شرعي تو قطاع الطريق
رزمجوئي کرد با چون من کسي
بس طبيبانند در بازار و کوي
که سگان خويشند با گرگان همه
چشم بسته پاي در چاهي نهاد
ساده و پاکيزه و زيبا و نرم
در شکسته حجره و ايوان سياه
گاه دستم گاه چشمم گاه گوش
پيش ما اي خواجه شکر پر بهاست
صد هزاران چون ترا کردم هلاک
پنبه اين شعله سوزان شديم
گفتم اين فکر محال از سر بنه
سوختم يک عمر و صبر آموختم
نيست اينجا جاي پيل و پيلبان
گندمم را ريختي تا زر دهي
چند ميگوئي ز جاه و مال و گنج
درزي ايام را اندازه نيست
گفت خامش کيسه شکر بجاست
اندرين آب گل آلود اي عجب
نه کمان آسايشي دارد نه تير
هم ز دزد و هم ز خانه بي‌خبر
ماکياني ساده از ده دور گشت
ديد موري در رهي پيلي سترگ
در زمستانم تف دل آتش است
رو بخواب امروز فردا نيز هست
خرمي کرديم وقت خرمي
آب و دان مور اندر جوي و جر
تند بادي کرد سيرش را تباه
گاه ره ميزد گهي ره ميسپرد
چند خندي بر در و ديوار من
جز به نان حرص کس فربه نشد
احتمال چاره‌جوئي دير بود
از براي ديگران بر منبريم
هر کجا کردم نگاه از پيش و پس
حيله روباهيش از ياد رفت
تا مگر پيراهني بخشد به وي
دزدي پنهان و پيدا کار توست
هيچگه ما را غم صياد نيست
